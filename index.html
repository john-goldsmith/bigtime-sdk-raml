BigTime SDK

Usage

Open your console and use window.sdk, which is an instance of BigTimeSdk.

Examples

      
window.sdk.resources.session.post({
  userid: '...', pwd: '...'
})
.then(
  response => {
    responseBody = JSON.parse(response.body);
    return window.sdk.resources.staff.get(null, {
      headers: {
        'x-auth-token': responseBody.token,
        'x-auth-realm': responseBody.firm
      }
    });
  },
  err => console.log('Error creating session:', err)
)
.then(
  response => {
    console.log(JSON.parse(response.body));
  },
  err => console.log('Error fetching staff listing:', err)
);
